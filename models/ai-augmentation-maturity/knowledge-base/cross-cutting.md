# Cross-Cutting Patterns — All Ecosystems

## AI-assisted PR descriptions

```yaml
id: cc_pr_descriptions
type: opportunity
ecosystem: cross-cutting
status: seed
discovered: 2026-03-30
updated: 2026-03-30

applies_when:
  - Active PR workflow (>5 PRs merged per month)
  - PR descriptions are inconsistent or thin
  - PR template exists but often not filled properly

value: MEDIUM
value_context: "Consistent PR descriptions improve review quality and historical traceability; AI can generate structured descriptions from diffs in seconds"
effort: Low
evidence_to_look_for:
  - PR template exists (.github/PULL_REQUEST_TEMPLATE.md)
  - Recent PRs with thin or no descriptions
  - High PR volume (>5/month)
seen_in: []

learning_entry: |
  Set up AI to draft PR descriptions from the diff before submitting.
  Template: What changed (bullet points), Why (link to issue), How to test,
  Breaking changes. Edit the draft — don't submit blindly.
  Tools: Claude Code generates these natively; GitHub Copilot has PR description feature.

readiness_criteria:
  - criterion: "PR template exists"
    type: Objective
    check: ".github/PULL_REQUEST_TEMPLATE.md exists"
  - criterion: "CI runs on PRs"
    type: Objective
    check: "CI workflow triggered on pull_request events"

kb_version: "6.2"

adoption_signals:
  active:
    - description: "CURSOR_SUMMARY markers with substantive content in PR descriptions"
      method: pr_analysis
      pattern: "PR body contains CURSOR_SUMMARY with >50 words of content"
      threshold: "count >= 3 in last 90 days"
    - description: "Copilot-generated PR summaries present"
      method: pr_analysis
      pattern: "PR body contains copilot:summary or Generated by Copilot markers"
      threshold: "count >= 3 in last 90 days"
    - description: "AI bot comments on PRs with description content"
      method: pr_analysis
      pattern: "PR comments from *[bot]* accounts containing summary or description content"
      threshold: "count >= 3 in last 90 days"
    - description: "CLAUDE.md references PR workflow or description generation"
      method: content_analysis
      pattern: "CLAUDE.md mentions PR descriptions, PR workflow, or PR template with >30 words"
      threshold: "exists: true"
  partial:
    - description: "1-2 PRs with AI-generated descriptions"
      method: pr_analysis
      pattern: "PR body contains CURSOR_SUMMARY or copilot:summary or AI bot comments"
      threshold: "count >= 1 AND count < 3 in last 90 days"
    - description: "AI config exists but no PR workflow reference"
      method: file_search
      pattern: "CLAUDE.md or .cursorrules or copilot-instructions.md exists"
      threshold: "exists: true"
      verification_hint: "File exists but may not reference PR workflow — partial infrastructure"
  absent:
    - description: "No AI-generated PR descriptions"
      method: pr_analysis
      pattern: "PR body with CURSOR_SUMMARY, copilot:summary, or AI bot description comments"
      threshold: "count == 0 in last 180 days"
    - description: "No AI config files"
      method: file_search
      pattern: "CLAUDE.md, .cursorrules, copilot-instructions.md, AGENTS.md"
      threshold: "none exist"
  anti_patterns:
    - description: "Empty CURSOR_SUMMARY markers (placeholder with no content)"
      method: pr_analysis
      pattern: "PR body contains CURSOR_SUMMARY marker but <10 words of content after it"
      actual_state: "Absent — empty markers indicate broken or abandoned setup"
    - description: "AI descriptions submitted without human review"
      method: pr_analysis
      pattern: "PR body is AI-generated boilerplate with no human edits (identical structure across 5+ PRs)"
      actual_state: "Anti-pattern — rubber-stamped AI output adds noise, not value"
  confidence_threshold: 60

readiness_levels:
  undiscovered:
    description: "No PR workflow or no PR template"
    criteria:
      - description: "No PR template"
        method: file_search
        pattern: ".github/PULL_REQUEST_TEMPLATE.md or .github/pull_request_template.md"
        threshold: "exists: false"
      - description: "No CI on pull requests"
        method: file_search
        pattern: "pull_request trigger in .github/workflows/*.yml"
        threshold: "count == 0"
  exploring:
    description: "PR workflow exists but descriptions are thin"
    requires: [undiscovered]
    criteria:
      - description: "PR template exists"
        method: file_search
        pattern: ".github/PULL_REQUEST_TEMPLATE.md"
        threshold: "exists: true"
      - description: "CI runs on PRs"
        method: file_search
        pattern: "pull_request trigger in .github/workflows/*.yml"
        threshold: "exists: true"
      - description: "Low PR volume or thin descriptions"
        method: pr_analysis
        pattern: "Merged PRs in last 90 days with body <50 words"
        threshold: "count >= 3"
    quantitative: "PR template exists, CI on PRs, but descriptions often thin"
  practiced:
    description: "Active PR workflow ready for AI-assisted descriptions"
    requires: [exploring]
    criteria:
      - description: "Active PR volume"
        method: pr_analysis
        pattern: "Merged PRs in last 30 days"
        threshold: "count >= 5"
      - description: "CI runs on PRs with status checks"
        method: file_search
        pattern: "pull_request trigger with required status checks in branch protection"
        threshold: "exists: true"
      - description: "PR template with structured sections"
        method: content_analysis
        pattern: ".github/PULL_REQUEST_TEMPLATE.md with ## sections (What, Why, How to test)"
        threshold: "section count >= 2"
    quantitative: ">5 PRs/month, CI on PRs, structured PR template"
  confidence_threshold: 60

ecosystem_overrides:
  typescript:
    adoption_signals:
      active:
        - description: "Cursor Bugbot comments on PRs"
          method: pr_analysis
          pattern: "PR comments from cursor-bugbot or Bugbot with code review content"
          threshold: "count >= 2 in last 90 days"
```

## CLAUDE.md with substantive project context

```yaml
id: cc_claude_md_context
type: opportunity
ecosystem: cross-cutting
status: seed
discovered: 2026-03-30
updated: 2026-03-30

applies_when:
  - Repo uses or plans to use AI tools (Claude, Copilot, Cursor)
  - No CLAUDE.md exists, or CLAUDE.md is generic/empty
  - Complex project with domain knowledge AI needs to work effectively

value: HIGH
value_context: "A substantive CLAUDE.md makes every AI interaction more effective — it's a one-time investment with compounding returns"
effort: Low
evidence_to_look_for:
  - CLAUDE.md absent
  - CLAUDE.md exists but <100 words or no project-specific content
  - CLAUDE.md missing coverage of: architecture, testing strategy, security-critical paths, build commands, coding conventions
seen_in:
  - repo: input-output-hk/lace-platform
    outcome: "comprehensive CLAUDE.md covering architecture, conventions, testing — measurably better AI interactions"

learning_entry: |
  Write a CLAUDE.md covering 6 categories:
  1. Architecture / module boundaries
  2. Coding conventions (naming, patterns, forbidden patterns)
  3. Testing strategy (what to test, how, which frameworks)
  4. Security-critical paths (what NOT to touch without review)
  5. Build and development commands
  6. Delivery workflow (branching, PR process, release)
  Start with one paragraph per category. Iterate as AI interactions reveal gaps.

readiness_criteria:
  - criterion: "AI tool in use or planned"
    type: Objective
    check: "Any AI config file exists, or AI-attributed commits present, or AI tool in dependencies"

kb_version: "6.2"

adoption_signals:
  active:
    - description: "CLAUDE.md exists with substantive content covering architecture and conventions"
      method: file_search
      pattern: "CLAUDE.md exists"
      threshold: "exists: true"
      verification_hint: "Must verify content quality — existence alone is not enough"
    - description: "CLAUDE.md covers architecture, conventions, and testing"
      method: content_analysis
      pattern: "CLAUDE.md contains >100 words AND references architecture/structure AND testing/conventions"
      threshold: "word count >= 100 AND topic coverage >= 3 of [architecture, conventions, testing, security, build, workflow]"
    - description: "CLAUDE.md referenced in commits or PRs"
      method: commit_scan
      pattern: "Commits touching CLAUDE.md in last 90 days"
      threshold: "count >= 2"
  partial:
    - description: "CLAUDE.md exists but thin (<100 words)"
      method: content_analysis
      pattern: "CLAUDE.md exists with <100 words"
      threshold: "word count >= 1 AND word count < 100"
    - description: "Alternative AI context file exists (not CLAUDE.md)"
      method: file_search
      pattern: ".cursorrules or copilot-instructions.md or AGENTS.md exists"
      threshold: "exists: true"
  absent:
    - description: "No AI context files"
      method: file_search
      pattern: "CLAUDE.md, .cursorrules, copilot-instructions.md, AGENTS.md, .claude/"
      threshold: "none exist"
    - description: "No AI attribution in recent commits"
      method: commit_scan
      pattern: "Co-authored-by.*(Claude|Copilot|Cursor|Gemini|AI)"
      threshold: "count == 0 in last 180 days"
  anti_patterns:
    - description: "Generic CLAUDE.md with no project-specific content"
      method: content_analysis
      pattern: "CLAUDE.md <100 words or contains only generic instructions (no repo-specific paths, no architecture, no conventions)"
      actual_state: "Absent — generic file provides no value and may mislead AI"
    - description: "CLAUDE.md exists but never updated since creation"
      method: commit_scan
      pattern: "CLAUDE.md has exactly 1 commit (initial creation) and repo has >50 commits since"
      actual_state: "Partial — stale context degrades over time"
  confidence_threshold: 60

readiness_levels:
  undiscovered:
    description: "No AI tools in use"
    criteria:
      - description: "No AI config files"
        method: file_search
        pattern: "CLAUDE.md, .cursorrules, copilot-instructions.md, AGENTS.md, .mcp.json"
        threshold: "none exist"
      - description: "No AI attribution in commits"
        method: commit_scan
        pattern: "Co-authored-by.*(Claude|Copilot|Cursor|Gemini|AI)"
        threshold: "count == 0 in last 180 days"
  exploring:
    description: "AI tools in use but no structured context"
    requires: [undiscovered]
    criteria:
      - description: "AI config or attribution present"
        method: file_search
        pattern: "CLAUDE.md, .cursorrules, copilot-instructions.md, AGENTS.md, or AI-attributed commits"
        threshold: "any exist"
      - description: "Project has meaningful complexity"
        method: file_search
        pattern: "Source files (*.ts, *.hs, *.rs, *.py, *.go, *.java)"
        threshold: "count >= 50"
    quantitative: "AI tools active, 50+ source files, no CLAUDE.md or thin CLAUDE.md"
  practiced:
    description: "Complex project with active AI use — ready for substantive CLAUDE.md"
    requires: [exploring]
    criteria:
      - description: "Active AI-attributed commits"
        method: commit_scan
        pattern: "Co-authored-by.*(Claude|Copilot|Cursor|Gemini|AI)"
        threshold: "count >= 5 in last 90 days"
      - description: "Multiple contributors"
        method: commit_scan
        pattern: "Distinct commit authors in last 90 days"
        threshold: "count >= 3"
      - description: "Complex project structure"
        method: file_search
        pattern: "Source files across 5+ directories"
        threshold: "directory count >= 5"
    quantitative: "5+ AI commits/90d, 3+ contributors, 5+ source directories"
  confidence_threshold: 60
```

## Trust boundary documentation (.aiignore)

```yaml
id: cc_aiignore_boundaries
type: opportunity
ecosystem: cross-cutting
status: seed
discovered: 2026-03-30
updated: 2026-03-30

applies_when:
  - High-assurance repo with security-critical code paths
  - AI tools in use without explicit trust boundaries
  - Crypto, consensus, financial, or authentication modules present

value: HIGH
value_context: "Explicit trust boundaries prevent AI from modifying security-critical code without human review — essential for high-assurance repos"
effort: Low
evidence_to_look_for:
  - .aiignore absent
  - Security-sensitive directories (crypto/, signing/, consensus/, auth/, keys/)
  - AI-attributed commits present but no .aiignore
seen_in: []

learning_entry: |
  Create .aiignore listing security-critical paths:
  - Cryptographic implementations
  - Consensus/protocol logic
  - Key management and signing
  - Financial state transitions
  - Authentication and authorization
  One path per line, same syntax as .gitignore.
  Review with security lead — they know which paths are critical.

readiness_criteria:
  - criterion: "Security-critical paths identifiable in file tree"
    type: Objective
    check: "Directories named crypto/, consensus/, auth/, signing/, or similar exist"
  - criterion: "AI tools in active use"
    type: Objective
    check: "AI-attributed commits or AI config files present"

kb_version: "6.2"

adoption_signals:
  active:
    - description: ".aiignore exists with meaningful path exclusions"
      method: file_search
      pattern: ".aiignore file exists"
      threshold: "exists: true"
    - description: ".aiignore contains >3 non-comment paths"
      method: content_analysis
      pattern: ".aiignore with non-comment, non-blank lines"
      threshold: "line count >= 3"
    - description: ".aiignore referenced in CLAUDE.md or onboarding docs"
      method: content_analysis
      pattern: "CLAUDE.md or CONTRIBUTING.md mentions .aiignore or trust boundaries"
      threshold: "exists: true"
  partial:
    - description: ".aiignore exists but minimal (1-2 paths)"
      method: content_analysis
      pattern: ".aiignore with non-comment, non-blank lines"
      threshold: "line count >= 1 AND line count < 3"
    - description: "Security paths mentioned in CLAUDE.md but no .aiignore"
      method: content_analysis
      pattern: "CLAUDE.md mentions security-critical paths or do-not-touch directories"
      threshold: "exists: true"
      verification_hint: "Informal boundary — better than nothing but not enforced"
  absent:
    - description: "No .aiignore file"
      method: file_search
      pattern: ".aiignore"
      threshold: "exists: false"
    - description: "No trust boundary documentation"
      method: file_search
      pattern: ".aiignore, or CLAUDE.md mentioning security boundaries"
      threshold: "none exist"
  anti_patterns:
    - description: ".aiignore that excludes everything (overly restrictive)"
      method: content_analysis
      pattern: ".aiignore contains wildcard patterns like * or **/* excluding >80% of repo"
      actual_state: "Anti-pattern — blocks AI from useful work, defeats the purpose"
    - description: ".aiignore with stale paths that no longer exist"
      method: content_analysis
      pattern: ".aiignore references paths not present in current file tree"
      actual_state: "Partial — unmaintained boundary erodes trust"
  confidence_threshold: 60

readiness_levels:
  undiscovered:
    description: "No security-sensitive paths or no AI tools"
    criteria:
      - description: "No security-critical directories"
        method: file_search
        pattern: "Directories named crypto/, auth/, consensus/, signing/, keys/, secrets/"
        threshold: "count == 0"
      - description: "No AI tools active"
        method: file_search
        pattern: "CLAUDE.md, .cursorrules, copilot-instructions.md, AGENTS.md"
        threshold: "none exist"
  exploring:
    description: "Security paths exist, AI tools active, no boundaries defined"
    requires: [undiscovered]
    criteria:
      - description: "Security-sensitive directories present"
        method: file_search
        pattern: "Directories named crypto/, auth/, consensus/, signing/, keys/, or similar"
        threshold: "count >= 1"
      - description: "AI tools active"
        method: file_search
        pattern: "CLAUDE.md, .cursorrules, copilot-instructions.md, or AI-attributed commits"
        threshold: "any exist"
      - description: "No .aiignore"
        method: file_search
        pattern: ".aiignore"
        threshold: "exists: false"
    quantitative: "Security paths present, AI active, no .aiignore"
  practiced:
    description: "Security-sensitive repo with AI tools — ready for trust boundaries"
    requires: [exploring]
    criteria:
      - description: "Multiple security-critical directories"
        method: file_search
        pattern: "Directories named crypto/, auth/, consensus/, signing/, keys/"
        threshold: "count >= 2"
      - description: "Active AI commits touching repo broadly"
        method: commit_scan
        pattern: "Co-authored-by.*(Claude|Copilot|Cursor|Gemini|AI)"
        threshold: "count >= 3 in last 90 days"
      - description: "AI config exists"
        method: file_search
        pattern: "CLAUDE.md or .cursorrules or copilot-instructions.md"
        threshold: "exists: true"
    quantitative: "2+ security dirs, 3+ AI commits/90d, AI config present"
  confidence_threshold: 60

ecosystem_overrides:
  haskell:
    readiness_levels:
      exploring:
        criteria:
          - description: "Haskell-specific security paths present"
            method: file_search
            pattern: "Rules/, conformance/, or formal-spec/ directories"
            threshold: "count >= 1"
  typescript:
    readiness_levels:
      exploring:
        criteria:
          - description: "TypeScript-specific security paths present"
            method: file_search
            pattern: "Packages or directories matching crypto*, signer*, auth-prompt*, or key-management*"
            threshold: "count >= 1"
```

## Onboarding documentation generation

```yaml
id: cc_onboarding_docs
type: opportunity
ecosystem: cross-cutting
status: seed
discovered: 2026-03-30
updated: 2026-03-30

applies_when:
  - Complex project with steep learning curve
  - CONTRIBUTING.md absent or thin
  - README focuses on usage, not development setup

value: MEDIUM
value_context: "Onboarding docs reduce time-to-contribution; AI can draft setup guides, architecture overviews, and contribution guides from existing code"
effort: Medium
evidence_to_look_for:
  - CONTRIBUTING.md absent or <50 words
  - README.md lacks development setup section
  - Complex build system (nix, multi-step setup)
  - New contributor PRs show common mistakes (indicates missing documentation)
seen_in: []

learning_entry: |
  Give Claude the repo structure + build config + CI workflow.
  Ask: "Draft a CONTRIBUTING.md covering: prerequisites, setup, build, test, PR process."
  Then: "Draft a development setup section for the README."
  Have a recent new contributor review the draft — they know what was confusing.

readiness_criteria:
  - criterion: "README exists with substantive content"
    type: Semi-objective
    check: "README.md >100 words of actual content"
  - criterion: "Build process is documented or scriptable"
    type: Semi-objective
    check: "Makefile, justfile, or build instructions in README/CONTRIBUTING"

kb_version: "6.2"

adoption_signals:
  active:
    - description: "AI-attributed commits touching CONTRIBUTING.md or onboarding docs"
      method: commit_scan
      pattern: "Co-authored-by.*(Claude|Copilot|Cursor|Gemini) in commits touching CONTRIBUTING.md, docs/onboarding*, or docs/getting-started*"
      threshold: "count >= 2 in last 90 days"
    - description: "AI-generated CONTRIBUTING.md with structured sections"
      method: content_analysis
      pattern: "CONTRIBUTING.md with >200 words AND ## sections covering setup/build/PR process"
      threshold: "section count >= 3 AND AI attribution in git log"
    - description: "AI-generated onboarding content in PRs"
      method: pr_analysis
      pattern: "PRs with AI co-authorship touching CONTRIBUTING.md, docs/onboarding*, or README development section"
      threshold: "count >= 1 in last 90 days"
  partial:
    - description: "CONTRIBUTING.md exists but not AI-assisted"
      method: file_search
      pattern: "CONTRIBUTING.md exists"
      threshold: "exists: true"
      verification_hint: "File exists but may predate AI tools — check if AI could improve it"
    - description: "AI config references onboarding or documentation"
      method: content_analysis
      pattern: "CLAUDE.md mentions CONTRIBUTING, onboarding, or documentation generation"
      threshold: "exists: true"
  absent:
    - description: "No CONTRIBUTING.md"
      method: file_search
      pattern: "CONTRIBUTING.md"
      threshold: "exists: false"
    - description: "No onboarding documentation"
      method: file_search
      pattern: "CONTRIBUTING.md, docs/onboarding*, docs/getting-started*, docs/setup*"
      threshold: "none exist"
  anti_patterns:
    - description: "AI-generated CONTRIBUTING.md with generic content not matching actual repo"
      method: content_analysis
      pattern: "CONTRIBUTING.md references tools, languages, or workflows not present in the repo"
      actual_state: "Anti-pattern — misleading docs are worse than no docs"
    - description: "Onboarding docs generated once and never maintained"
      method: commit_scan
      pattern: "CONTRIBUTING.md has 1 commit and repo has >100 commits since"
      actual_state: "Partial — stale onboarding docs create confusion for new contributors"
  confidence_threshold: 60

readiness_levels:
  undiscovered:
    description: "No documentation infrastructure"
    criteria:
      - description: "No README or minimal README"
        method: content_analysis
        pattern: "README.md word count"
        threshold: "word count < 100 OR exists: false"
      - description: "No CONTRIBUTING.md"
        method: file_search
        pattern: "CONTRIBUTING.md"
        threshold: "exists: false"
  exploring:
    description: "README exists but onboarding docs thin or absent"
    requires: [undiscovered]
    criteria:
      - description: "README with substantive content"
        method: content_analysis
        pattern: "README.md word count"
        threshold: "word count >= 100"
      - description: "CONTRIBUTING.md absent or thin"
        method: content_analysis
        pattern: "CONTRIBUTING.md word count"
        threshold: "exists: false OR word count < 100"
      - description: "Complex build process"
        method: file_search
```
